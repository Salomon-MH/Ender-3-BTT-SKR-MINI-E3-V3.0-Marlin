; Ender 3 Custom Start G-code
G92 E0 ; Reset Extruder

; Heat bed to target temperature
M140 S{material_bed_temperature_layer_0} ; set bed temperature dont wait
M105 ; request temp report
M190 S{material_bed_temperature_layer_0} ; set bed temperature and wait heat up

; Custom pre-heat for warm nozzle when auto-leveling
M104 S175 ; set nozzle temperature dont wait
M105 ; request temp report
M109 S175 ; set nozzle temperature and wait heat up

G28 ; Home all axes
G29 ; Auto Bed Leveling

; Move to beginning to prevent leakage off right side of the bed
G1 Z2.0 F3000 ; Move Z Axis up little to prevent scratching of Heat Bed
G1 X0.1 Y10 Z3.0 F5000.0 ; Move to start position

; Heat nozzle to target temperature
M104 S{material_print_temperature_layer_0} ; set nozzle temperature dont wait
M105 ; request temp report
M109 S{material_print_temperature_layer_0} ; set nozzle temperature and wait heat up

; Startup routine
G1 Z2.0 F3000 ; Move Z Axis up little to prevent scratching of Heat Bed
G1 X0.1 Y20 Z0.3 F5000.0 ; Move to start position
G1 X0.1 Y200.0 Z0.3 F1500.0 E15 ; Draw the first line
G1 X0.4 Y200.0 Z0.3 F5000.0 ; Move to side a little
G1 X0.4 Y20 Z0.3 F1500.0 E30 ; Draw the second line
G92 E0 ; Reset Extruder
G1 Z2.0 F3000 ; Move Z Axis up little to prevent scratching of Heat Bed
G1 X5 Y20 Z0.3 F5000.0 ; Move over to prevent blob squish
